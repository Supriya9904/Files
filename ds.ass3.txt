#include <mpi.h>
#include <stdio.h>
#include <stdlib.h>

int main(int argc, char *argv[]) {

    MPI_Init(&argc, &argv);

    int rank, size;
    MPI_Comm_rank(MPI_COMM_WORLD, &rank);
    MPI_Comm_size(MPI_COMM_WORLD, &size);

    int unitSize = 5;
    int root = 0;

    int total_elements = unitSize * size;

    int *send_buffer = NULL;
    int receive_buffer[5];
    int gather_buffer[size];

    if (rank == root) {
        send_buffer = (int *)malloc(total_elements * sizeof(int));

        printf("Enter %d elements:\n", total_elements);
        for (int i = 0; i < total_elements; i++) {
            scanf("%d", &send_buffer[i]);
        }
    }

    MPI_Scatter(
        send_buffer, unitSize, MPI_INT,
        receive_buffer, unitSize, MPI_INT,
        root, MPI_COMM_WORLD
    );

    int local_sum = 0;
    for (int i = 0; i < unitSize; i++) {
        local_sum += receive_buffer[i];
    }

    printf("Intermediate sum at process %d = %d\n", rank, local_sum);

    MPI_Gather(
        &local_sum, 1, MPI_INT,
        gather_buffer, 1, MPI_INT,
        root, MPI_COMM_WORLD
    );

    if (rank == root) {
        int total_sum = 0;
        for (int i = 0; i < size; i++) {
            total_sum += gather_buffer[i];
        }

        printf("Final Sum = %d\n", total_sum);
    }

    MPI_Finalize();
    return 0;
}
