from timeit import default_timer as timer
from dateutil import parser
import threading
import datetime
import socket
import time

# send time to server
def startSendingTime(slave_client):
    while True:
        slave_client.send(str(datetime.datetime.now()).encode())
        print("Recent time sent successfully\n")
        time.sleep(5)

# receive synchronized time
def startReceivingTime(slave_client):
    while True:
        synchronized_time = parser.parse(slave_client.recv(1024).decode())
        print("Synchronized time at the client is:", synchronized_time, "\n")

# start client
def initiateSlaveClient(port=8080):
    slave_client = socket.socket()
    slave_client.connect(('127.0.0.1', port))

    print("Starting to receive time from server\n")

    send_thread = threading.Thread(target=startSendingTime, args=(slave_client,))
    send_thread.start()

    print("Starting to receiving synchronized time from server\n")

    receive_thread = threading.Thread(target=startReceivingTime, args=(slave_client,))
    receive_thread.start()

# main
if __name__ == '__main__':
    initiateSlaveClient(port=8080)

from dateutil import parser
import threading
import datetime
import socket
import time

# store client data
client_data = {}

# receive time from client
def startReceivingClockTime(connector, address):
    while True:
        clock_time_string = connector.recv(1024).decode()
        clock_time = parser.parse(clock_time_string)

        clock_time_diff = datetime.datetime.now() - clock_time

        client_data[address] = {
            "clock_time": clock_time,
            "time_difference": clock_time_diff,
            "connector": connector
        }

        print("Client Data updated with:", address, "\n")
        time.sleep(5)

# accept clients
def startConnecting(master_server):
    while True:
        conn, addr = master_server.accept()
        address = str(addr[0]) + ":" + str(addr[1])

        print(address + " connected")

        thread = threading.Thread(
            target=startReceivingClockTime,
            args=(conn, address)
        )
        thread.start()

# calculate average difference
def getAverageClockDiff():
    time_diffs = [client["time_difference"] for client in client_data.values()]
    total = sum(time_diffs, datetime.timedelta(0, 0))
    avg = total / len(client_data)
    return avg

# synchronize clocks
def synchronizeAllClocks():
    while True:
        print("New synchronization cycle started.")
        print("Clients:", len(client_data))

        if len(client_data) > 0:
            avg_diff = getAverageClockDiff()

            for addr, client in client_data.items():
                try:
                    sync_time = datetime.datetime.now() + avg_diff
                    client["connector"].send(str(sync_time).encode())
                except:
                    print("Error sending to", addr)
        else:
            print("No client data. Synchronization not possible.")

        print("\n")
        time.sleep(5)

# start server
def initiateClockServer(port=8080):
    server = socket.socket()
    server.setsockopt(socket.SOL_SOCKET, socket.SO_REUSEADDR, 1)

    print("Server socket created\n")

    server.bind(('', port))
    server.listen(10)

    print("Clock server started...\n")

    threading.Thread(target=startConnecting, args=(server,)).start()
    threading.Thread(target=synchronizeAllClocks).start()

# main
if __name__ == '__main__':
    initiateClockServer(port=8080)

TokenRing.java
import java.util.*;

public class TokenRing {
    public static void main(String args[]) {

        Scanner scan = new Scanner(System.in);

        System.out.println("Enter number of nodes:");
        int n = scan.nextInt();

        int token = 0;
        int ch = 0, flag = 0;

        // Display ring
        System.out.print("Ring: ");
        for (int i = 0; i < n; i++) {
            System.out.print(" " + i);
        }
        System.out.println(" " + 0);

        do {
            System.out.println("\nEnter sender:");
            int s = scan.nextInt();

            System.out.println("Enter receiver:");
            int r = scan.nextInt();

            System.out.println("Enter data:");
            int data = scan.nextInt();

            // Token passing
            System.out.print("Token passing: ");
            for (int i = token; (i % n) != s; i++) {
                System.out.print((i % n) + " -> ");
            }
            System.out.println(s);

            System.out.println("Sender " + s + " sending data: " + data);

            // Data forwarding
            for (int i = (s + 1) % n; i != r; i = (i + 1) % n) {
                System.out.println("Data " + data + " forwarded by " + i);
            }

            System.out.println("Receiver " + r + " received data: " + data);

            token = s;

            // Continue or not
            do {
                if (flag == 1)
                    System.out.println("Invalid Input!!");

                System.out.print("Do you want to send again? (1=Yes, 0=No): ");
                ch = scan.nextInt();

                if (ch != 0 && ch != 1)
                    flag = 1;
                else
                    flag = 0;

            } while (ch != 0 && ch != 1);

        } while (ch == 1);

        scan.close();
    }
}

?? PART 3: Simple Theory (Exam Ready)
?? What is Token Ring Algorithm?
* It is used for mutual exclusion in distributed systems 
* Only one process can enter critical section at a time 

?? How it works
1. All processes form a logical ring 
2. A token is passed around the ring 
3. Only process with token can:
?? enter critical section 
4. After work ? token passed to next process 

?? Real-life Example
?? Think:
* 5 students sitting in circle 
* One pen ?? (token) 
* Only student with pen can write 

?? Key Points (from your PDF page 3)
* Only one token exists ? ensures mutual exclusion 
* No deadlock 
* No starvation 
* Fair execution order 

?? Program Logic (Simple)
StepWhat happensInput nodesCreate ringSender chosenNeeds tokenToken travelsFrom current ? senderSender sends dataStarts communicationData forwardedThrough ringReceiver gets dataProcess ends

Ass 6: Implement Bully algorithm for leader election
Bully.java
import java.util.*;

public class Bully {

    int coordinator;
    int max_processes;
    boolean processes[];

    // Constructor
    public Bully(int max) {
        max_processes = max;
        processes = new boolean[max_processes];

        coordinator = max; // highest process is coordinator

        System.out.println("Creating processes...");
        for (int i = 0; i < max; i++) {
            processes[i] = true;
            System.out.println("P" + (i + 1) + " created");
        }

        System.out.println("Process P" + coordinator + " is the coordinator");
    }

    // Display processes
    void displayProcesses() {
        for (int i = 0; i < max_processes; i++) {
            if (processes[i]) {
                System.out.println("P" + (i + 1) + " is UP");
            } else {
                System.out.println("P" + (i + 1) + " is DOWN");
            }
        }
        System.out.println("Coordinator is P" + coordinator);
    }

    // Bring process up
    void upProcess(int process_id) {
        if (!processes[process_id - 1]) {
            processes[process_id - 1] = true;
            System.out.println("Process P" + process_id + " is now UP");
        } else {
            System.out.println("Process already UP");
        }
    }

    // Bring process down
    void downProcess(int process_id) {
        if (!processes[process_id - 1]) {
            System.out.println("Process already DOWN");
        } else {
            processes[process_id - 1] = false;
            System.out.println("Process P" + process_id + " is DOWN");
        }
    }

    // Election Algorithm
    void runElection(int process_id) {
        coordinator = process_id;

        for (int i = process_id; i < max_processes; i++) {
            System.out.println("Election message sent from P" + process_id + " to P" + (i + 1));

            if (processes[i]) {
                runElection(i + 1);
                return;
            }
        }

        System.out.println("Process P" + coordinator + " becomes coordinator");
    }

    // Main method
    public static void main(String args[]) {

        Scanner sc = new Scanner(System.in);
        Bully bully = null;

        int choice, max_processes, process_id;

        while (true) {

            System.out.println("\n--- Bully Algorithm ---");
            System.out.println("1. Create processes");
            System.out.println("2. Display processes");
            System.out.println("3. Up a process");
            System.out.println("4. Down a process");
            System.out.println("5. Run election");
            System.out.println("6. Exit");

            System.out.print("Enter choice: ");
            choice = sc.nextInt();

            switch (choice) {

                case 1:
                    System.out.print("Enter number of processes: ");
                    max_processes = sc.nextInt();
                    bully = new Bully(max_processes);
                    break;

                case 2:
                    if (bully != null)
                        bully.displayProcesses();
                    break;

                case 3:
                    System.out.print("Enter process number to UP: ");
                    process_id = sc.nextInt();
                    bully.upProcess(process_id);
                    break;

                case 4:
                    System.out.print("Enter process number to DOWN: ");
                    process_id = sc.nextInt();
                    bully.downProcess(process_id);
                    break;

                case 5:
                    System.out.print("Enter process to start election: ");
                    process_id = sc.nextInt();
                    bully.runElection(process_id);
                    bully.displayProcesses();
                    break;

                case 6:
                    System.exit(0);

                default:
                    System.out.println("Invalid choice");
            }
        }
    }
}
